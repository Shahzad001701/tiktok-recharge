<script>
    // JavaScript Logic
    const rechargeButton = document.getElementById('rechargeButton');
    const paymentModal = document.getElementById('paymentModal');
    const processingDiv = document.getElementById('processingDiv');
    const successDiv = document.getElementById('successDiv');
    const finalCoinsDisplay = document.getElementById('finalCoins');
    const packages = document.querySelectorAll('.package-item');
    const totalPriceDisplay = document.getElementById('totalPriceDisplay');
    
    // Set default selected coins based on the largest package (17,500)
    let selectedCoins = 17500; 
    let selectedPrice = "224.99";

    // Handle package selection
    packages.forEach(pkg => {
        pkg.addEventListener('click', () => {
            packages.forEach(p => p.classList.remove('selected'));
            pkg.classList.add('selected');
            selectedCoins = pkg.getAttribute('data-coins');
            selectedPrice = pkg.getAttribute('data-price');
            
            if (selectedCoins === 'custom') {
                // Faking a large custom amount and price for the demo
                selectedCoins = 50000;
                selectedPrice = '528.60';
                totalPriceDisplay.textContent = 'Total: $' + selectedPrice;
            } else {
                 totalPriceDisplay.textContent = 'Total: $' + selectedPrice;
            }
        });
    });

    // Handle Recharge Button Click
    rechargeButton.addEventListener('click', () => {
        // Show Payment Processing Modal
        paymentModal.style.display = 'flex';
        processingDiv.style.display = 'block';
        successDiv.style.display = 'none';
        
        // Format coins with comma (e.g., 17,500)
        finalCoinsDisplay.textContent = selectedCoins.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ","); 

        // Simulate Payment Processing Time (3 seconds)
        setTimeout(() => {
            // Show Success Message
            processingDiv.style.display = 'none';
            successDiv.style.display = 'block';
        }, 3000); 
    });

</script>
